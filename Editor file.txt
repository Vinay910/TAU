package com.LargestPrime;

public class LargestPrime {
	
	public static int getLargestPrime(int number)
	{
		if(number>0)
		{
			int temp=0;
			for(int i=1;i<number;i++)
			{
				if(number%i==0)
				{
					temp=i;
				}
			}
			if(temp==1)
			{
				return number;
			}
			return temp;
		}
		else
		{
			return -1;
		}
	}

}
